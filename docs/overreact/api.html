 40def get_internal_energies(
 41    compounds: dict, qrrho: bool = True, temperature: float = 298.15
 42):
 43    """Obtain internal energies for compounds at a given temperature.
 44
 45    Parameters
 46    ----------
 47    compounds : dict-like
 48        A descriptor of the compounds.
 49        Mostly likely, this comes from a parsed input file.
 50        See `overreact.io.parse_model`.
 51    qrrho : bool, optional
 52        Apply the quasi-rigid rotor harmonic oscillator (QRRHO) approximation of
 53        M. Head-Gordon and others (see
 54        [*J. Phys. Chem. C* **2015**, 119, 4, 1840–1850](http://dx.doi.org/10.1021/jp509921r))
 55        on top of the classical RRHO.
 56    temperature : array-like, optional
 57        Absolute temperature in Kelvin.
 58
 59    Returns
 60    -------
 61    array-like
 62
 63    Examples
 64    --------
 65    >>> import overreact as rx
 66    >>> from overreact import _constants as constants
 67    >>> model = rx.parse_model("data/ethane/B97-3c/model.k")
 68    >>> internal_energies = get_internal_energies(model.compounds)
 69    >>> (internal_energies - internal_energies.min()) / constants.kcal
 70    array([0. , 2.20053981])
 71
 72    """
 73    compounds = rx.io._check_compounds(compounds)
 74    internal_energies = []
 75    for name in compounds:
 76        logger.info(f"calculate internal energy: {name}")
 77
 78        # TODO(schneiderfelipe): inertia might benefit from caching
 79        moments, _, _ = coords.inertia(
 80            compounds[name].atommasses, compounds[name].atomcoords
 81        )
 82
 83        internal_energy = rx.thermo.calc_internal_energy(
 84            energy=compounds[name].energy,
 85            degeneracy=compounds[name].mult,
 86            moments=moments,
 87            vibfreqs=compounds[name].vibfreqs,
 88            qrrho=qrrho,
 89            temperature=temperature,
 90        )
 91        internal_energies.append(internal_energy)
 92    return np.array(internal_energies)

 95def get_enthalpies(compounds: dict, qrrho: bool = True, temperature: float = 298.15):
 96    """Obtain enthalpies for compounds at a given temperature.
 97
 98    Parameters
 99    ----------
100    compounds : dict-like
101        A descriptor of the compounds.
102        Mostly likely, this comes from a parsed input file.
103        See `overreact.io.parse_model`.
104    qrrho : bool, optional
105        Apply the quasi-rigid rotor harmonic oscillator (QRRHO) approximation of
106        M. Head-Gordon and others (see
107        [*J. Phys. Chem. C* **2015**, 119, 4, 1840–1850](http://dx.doi.org/10.1021/jp509921r))
108        on top of the classical RRHO.
109    temperature : array-like, optional
110        Absolute temperature in Kelvin.
111
112    Returns
113    -------
114    array-like
115
116    Examples
117    --------
118    >>> import overreact as rx
119    >>> from overreact import _constants as constants
120    >>> model = rx.parse_model("data/ethane/B97-3c/model.k")
121    >>> enthalpies = get_enthalpies(model.compounds)
122    >>> (enthalpies - enthalpies.min()) / constants.kcal
123    array([0. , 2.20053981])
124
125    The enthalpies at absolute zero can easily be obtained (this is used,
126    e.g., in the calculation of the Eckart tunneling coefficient, see
127    `overreact.tunnel.eckart`). We can use this to calculate, for instance,
128    the thermal contributions to the enthalpy:
129
130    >>> zero_enthalpies = get_enthalpies(model.compounds, temperature=0)
131    >>> (enthalpies - zero_enthalpies) / constants.kcal
132    array([2.78, 2.50])
133    """
134    compounds = rx.io._check_compounds(compounds)
135    enthalpies = []
136    for name in compounds:
137        logger.info(f"calculate enthalpy: {name}")
138
139        # TODO(schneiderfelipe): inertia might benefit from caching
140        moments, _, _ = coords.inertia(
141            compounds[name].atommasses, compounds[name].atomcoords
142        )
143
144        enthalpy = rx.thermo.calc_enthalpy(
145            energy=compounds[name].energy,
146            degeneracy=compounds[name].mult,
147            moments=moments,
148            vibfreqs=compounds[name].vibfreqs,
149            qrrho=qrrho,
150            temperature=temperature,
151        )
152        enthalpies.append(enthalpy)
153    return np.array(enthalpies)

156def get_entropies(
157    compounds: dict,
158    environment: str | None = None,
159    method: str = "standard",
160    qrrho: bool = True,
161    temperature: float = 298.15,
162    pressure: float = constants.atm,
163):
164    """Obtain entropies for compounds at a given temperature and pressure.
165
166    Parameters
167    ----------
168    compounds : dict-like
169        A descriptor of the compounds.
170        Mostly likely, this comes from a parsed input file.
171        See `overreact.io.parse_model`.
172    environment : str or None, optional
173        Choose between "gas" and a solvent. This is chosen for you by default,
174        based on the names of each compound (e.g. `A(g)` or `A` is gas,
175        `A(w)` or `A(...)` is solvated). In case this is given, all compounds
176        will have the same behavior.
177    method : str, optional
178        This is a placeholder for future functionality.
179        There are plans to implement more sophisticated methods for calculating
180        entropies such as in
181        [*Phys. Chem. Chem. Phys.*, **2019**, 21, 18920-18929](https://doi.org/10.1039/C9CP03226F)
182        and
183        [*J. Chem. Theory Comput.* **2019**, 15, 5, 3204–3214](https://doi.org/10.1021/acs.jctc.9b00214).
184        Head over to the
185        [discussions](https://github.com/geem-lab/overreact/discussions) if
186        you're interested and would like to contribute.
187        Leave this as "standard" for now.
188    qrrho : bool, optional
189        Apply the quasi-rigid rotor harmonic oscillator (QRRHO) approximation of
190        S. Grimme (see
191        [*Theory. Chem. Eur. J.*, **2012**, 18: 9955-9964](https://doi.org/10.1002/chem.201200497))
192        on top of the classical RRHO.
193    temperature : array-like, optional
194        Absolute temperature in Kelvin.
195    pressure : array-like, optional
196        Reference gas pressure.
197
198    Returns
199    -------
200    array-like
201
202    Examples
203    --------
204    >>> import overreact as rx
205    >>> from overreact import _constants as constants
206    >>> model = rx.parse_model("data/ethane/B97-3c/model.k")
207    >>> entropies = get_entropies(model.compounds)
208    >>> (entropies - entropies.min()) / constants.calorie
209    array([1.4, 0. ])
210
211    You can consider all compounds as solvated if you want:
212
213    >>> sol_entropies = get_entropies(model.compounds, environment="solvent")
214    >>> (sol_entropies - entropies) / constants.calorie
215    array([-6.35360874, -6.35360874])
216    """
217    compounds = rx.io._check_compounds(compounds)
218    entropies = []
219    for name in compounds:
220        logger.info(f"calculate entropy: {name}")
221
222        if "point_group" in compounds[name]:
223            point_group = compounds[name].point_group
224        else:
225            point_group = coords.find_point_group(
226                compounds[name].atommasses, compounds[name].atomcoords
227            )
228        symmetry_number = coords.symmetry_number(point_group)
229
230        # TODO(schneiderfelipe): inertia might benefit from caching
231        moments, _, _ = coords.inertia(
232            compounds[name].atommasses, compounds[name].atomcoords
233        )
234
235        if environment is None:
236            environment = rx.core._get_environment(name)
237        entropy = rx.thermo.calc_entropy(
238            atommasses=compounds[name].atommasses,
239            atomcoords=compounds[name].atomcoords,
240            energy=compounds[name].energy,
241            degeneracy=compounds[name].mult,
242            moments=moments,
243            symmetry_number=symmetry_number,
244            vibfreqs=compounds[name].vibfreqs,
245            environment=environment,
246            method=method,
247            qrrho=qrrho,
248            temperature=temperature,
249            pressure=pressure,
250        )
251
252        if compounds[name].symmetry is not None:
253            # The negative sign here seems correct. See equations (9) and (10)
254            # of doi:10.1002/qua.25686.
255            entropy -= rx.change_reference_state(
256                compounds[name].symmetry,
257                1,
258                temperature=temperature,
259                pressure=pressure,
260            )
261
262        entropies.append(entropy)
263    return np.array(entropies)

266def _check_qrrho(qrrho: bool | tuple[bool, bool]) -> tuple[bool, bool]:
267    """Get options for QRRHO for both enthalpy and entropy.
268
269    Parameters
270    ----------
271    qrrho : bool or tuple-like
272        Apply both the quasi-rigid rotor harmonic oscillator (QRRHO)
273        approximations of M. Head-Gordon and others (enthalpy correction, see
274        [*J. Phys. Chem. C* **2015**, 119, 4, 1840–1850](http://dx.doi.org/10.1021/jp509921r))
275        and S. Grimme (entropy correction, see
276        [*Theory. Chem. Eur. J.*, **2012**, 18: 9955-9964](https://doi.org/10.1002/chem.201200497))
277        on top of the classical RRHO.
278
279    Returns
280    -------
281    qrrho_enthalpy, qrrho_entropy : bool
282
283    Raises
284    ------
285    ValueError
286        If the QRRHO specification is not understood.
287
288    Examples
289    --------
290    >>> _check_qrrho(True)
291    (True, True)
292    >>> _check_qrrho(False)
293    (False, False)
294    >>> _check_qrrho((True, False))
295    (True, False)
296    >>> _check_qrrho((False, True))
297    (False, True)
298    """
299    if qrrho is True:
300        return True, True
301    elif qrrho is False:
302        return False, False
303    elif isinstance(qrrho, tuple):
304        return qrrho
305    else:
306        raise ValueError(f"unrecognized QRRHO specification: {qrrho}")

309def get_freeenergies(
310    compounds: dict,
311    bias: float = 0.0,
312    environment: str | None = None,
313    method: str = "standard",
314    qrrho: bool | tuple[bool, bool] = True,
315    temperature: float = 298.15,
316    pressure: float = constants.atm,
317):
318    """Obtain free energies for compounds at a given temperature and pressure.
319
320    Parameters
321    ----------
322    compounds : dict-like
323        A descriptor of the compounds.
324        Mostly likely, this comes from a parsed input file.
325        See `overreact.io.parse_model`.
326    bias : array-like, optional
327        Energy to be added to free energies.
328    environment : str or None, optional
329        Choose between "gas" and a solvent. This is chosen for you by default,
330        based on the names of each compound. If given, all compounds will
331        have the same behavior.
332    method : str, optional
333        This is a placeholder for future functionality.
334        There are plans to implement more sophisticated methods for calculating
335        entropies such as in
336        [*Phys. Chem. Chem. Phys.*, **2019**, 21, 18920-18929](https://doi.org/10.1039/C9CP03226F)
337        and
338        [*J. Chem. Theory Comput.* **2019**, 15, 5, 3204–3214](https://doi.org/10.1021/acs.jctc.9b00214).
339        Head over to the
340        [discussions](https://github.com/geem-lab/overreact/discussions) if
341        you're interested and would like to contribute.
342        Leave this as "standard" for now.
343    qrrho : bool or tuple-like, optional
344        Apply both the quasi-rigid rotor harmonic oscillator (QRRHO)
345        approximations of M. Head-Gordon and others (enthalpy correction, see
346        [*J. Phys. Chem. C* **2015**, 119, 4, 1840–1850](http://dx.doi.org/10.1021/jp509921r))
347        and S. Grimme (entropy correction, see
348        [*Theory. Chem. Eur. J.*, **2012**, 18: 9955-9964](https://doi.org/10.1002/chem.201200497))
349        on top of the classical RRHO.
350    temperature : array-like, optional
351        Absolute temperature in Kelvin.
352    pressure : array-like, optional
353        Reference gas pressure.
354
355    Returns
356    -------
357    array-like
358
359    Examples
360    --------
361    >>> import overreact as rx
362    >>> from overreact import _constants as constants
363    >>> model = rx.parse_model("data/ethane/B97-3c/model.k")
364    >>> freeenergies = get_freeenergies(model.compounds, qrrho=(False, True))
365    >>> (freeenergies - freeenergies.min()) / constants.kcal
366    array([0. , 2.62281461])
367    >>> freeenergies = get_freeenergies(model.compounds)
368    >>> (freeenergies - freeenergies.min()) / constants.kcal
369    array([0. , 2.62862818])
370
371    You can consider all compounds as solvated if you want:
372
373    >>> sol_freeenergies = get_freeenergies(model.compounds, environment="solvent")
374    >>> (sol_freeenergies - freeenergies) / constants.kcal
375    array([1.89432845, 1.89432845])
376
377    You can set a simple energy bias, either as a constant or compound-wise:
378
379    >>> get_freeenergies(model.compounds, bias=1.0) - freeenergies
380    array([1., 1.])
381    >>> get_freeenergies(model.compounds, bias=-1.0) - freeenergies
382    array([-1., -1.])
383    >>> get_freeenergies(model.compounds, bias=[1.0, -1.0]) - freeenergies
384    array([ 1., -1.])
385    """
386    qrrho_enthalpy, qrrho_entropy = _check_qrrho(qrrho)
387    enthalpies = get_enthalpies(
388        compounds, qrrho=qrrho_enthalpy, temperature=temperature
389    )
390    entropies = get_entropies(
391        compounds,
392        environment=environment,
393        method=method,
394        qrrho=qrrho_entropy,
395        temperature=temperature,
396        pressure=pressure,
397    )
398    # TODO(schneiderfelipe): log the contribution of bias
399    return enthalpies - temperature * entropies + np.asarray(bias)

402def get_k(
403    scheme: Scheme,
404    compounds: dict | None = None,
405    bias: float = 0.0,
406    tunneling: str = "eckart",
407    qrrho: bool | tuple[bool, bool] = True,
408    scale: str = "l mol-1 s-1",
409    temperature: float = 298.15,
410    pressure: float = constants.atm,
411    delta_freeenergies: float | None = None,
412    molecularity: float | None = None,
413    volume: float | None = None,
414) -> float:
415    r"""Obtain reaction rate constants for a given reaction scheme.
416
417    Parameters
418    ----------
419    scheme : Scheme
420        A descriptor of the reaction scheme.
421        Mostly likely, this comes from a parsed input file.
422        See `overreact.io.parse_model`.
423    compounds : dict-like, optional
424        A descriptor of the compounds.
425        Mostly likely, this comes from a parsed input file.
426        See `overreact.io.parse_model`.
427    bias : array-like, optional
428        Energy to be added to free energies.
429    tunneling : str or None, optional
430        Choose between "eckart", "wigner" or None (or "none").
431    qrrho : bool or tuple-like, optional
432        Apply both the quasi-rigid rotor harmonic oscillator (QRRHO)
433        approximations of M. Head-Gordonand others (enthalpy correction, see
434        [*J. Phys. Chem. C* **2015**, 119, 4, 1840–1850](http://dx.doi.org/10.1021/jp509921r))
435        and S. Grimme (entropy correction, see
436        [*Theory. Chem. Eur. J.*, **2012**, 18: 9955-9964](https://doi.org/10.1002/chem.201200497))
437        on top of the classical RRHO.
438    scale : str, optional
439        Reaction rate units. Possible values are "cm3 mol-1 s-1",
440        "l mol-1 s-1", "m3 mol-1 s-1", "cm3 particle-1 s-1", "mmHg-1 s-1" and
441        "atm-1 s-1".
442    temperature : array-like, optional
443        Absolute temperature in Kelvin.
444    pressure : array-like, optional
445        Reference gas pressure.
446    delta_freeenergies : array-like, optional
447        Use this instead of obtaining delta free energies from the compounds.
448    molecularity : array-like, optional
449        Reaction order, i.e., number of molecules that come together to react.
450        If set, this is used to calculate `delta_moles` for
451        `overreact.thermo.equilibrium_constant`, which effectively calculates a solution
452        equilibrium constant between reactants and the transition state for
453        gas phase data. You should set this to `None` if your free energies
454        were already adjusted for solution Gibbs free energies.
455    volume : float, optional
456        Molar volume.
457
458    Returns
459    -------
460    array-like
461
462    Notes
463    -----
464    Some symbols are accepted as alternatives in "scale": "M-1", "ml" and
465    "torr-1" are understood as "l mol-1", "cm3" and "mmHg-1", respectively.
466
467    Examples
468    --------
469    Below is an example of an estimate for the rate of methyl rotation in
470    ethane (a trivial attempt to reproduce
471    [*Science*, **2006**, 313, 5795, 1951-1955](https://doi.org/10.1126/science.1132178)).
472    **How many turns it does per second?**
473
474    >>> model = rx.parse_model("data/ethane/B97-3c/model.k")
475    >>> get_k(model.scheme, model.compounds)
476    array([8.16e+10])
477    >>> get_k(model.scheme, model.compounds, qrrho=(False, True))
478    array([8.24968117e+10])
479    >>> get_k(model.scheme, model.compounds, qrrho=False)
480    array([8.26909266e+10])
481    >>> get_k(model.scheme, model.compounds, tunneling="wigner")
482    array([7.99e+10])
483    >>> get_k(model.scheme, model.compounds, tunneling=None)
484    array([7.35e+10])
485
486    The calculated value is off by less than 2% from the experimental value
487    (:math:`\frac{1}{12 \times 10^{-12}} \text{s}^{-1} = 8.33 \times 10^{10} \text{s}^{-1}`).
488    We use Eckart tunneling by default, but see the effect of changing it
489    above.
490
491    The units of the returned reaction rate constants can be selected for
492    non-unimolecular processes. The following is an attempt to reproduce
493    [*J Atmos Chem*, **1996** 23, 37–49](https://doi.org/10.1007/BF00058703) for
494    the reaction of proton-withdrawal by a chloride radical from the methane
495    molecule
496    :math:`\ce{CH4 + \cdot Cl -> [H3C\cdots H\cdots Cl]^\ddagger -> H3C\cdot + HCl}`:
497
498    >>> model = rx.parse_model("data/tanaka1996/UMP2/cc-pVTZ/model.jk")
499    >>> get_k(model.scheme, model.compounds, temperature=300,
500    ...       scale="cm3 particle-1 s-1")
501    array([9.60e-14])
502
503    (By the way, according to the Jet Propulsion Laboratory,
504    [Publication No. 19-5](https://jpldataeval.jpl.nasa.gov/),
505    the experimental reaction rate constant for this reaction is
506    :math:`1.0 \times 10^{-13} \text{cm}^3 \text{particle}^{-1} \text{s}^{-1}`.)
507
508    The returned units are "M-1 s-1" by default:
509
510    >>> get_k(model.scheme, model.compounds) \
511    ... == get_k(model.scheme, model.compounds, scale="l mol-1 s-1")
512    array([ True])
513
514    You can also turn the tunneling correction off by using the string "none":
515
516    >>> get_k(model.scheme, model.compounds, tunneling="none") \
517    ... == get_k(model.scheme, model.compounds, tunneling=None)
518    array([ True])
519
520    You can set a simple energy bias, either as a constant or compound-wise:
521
522    >>> get_k(model.scheme, model.compounds, bias=1.0 * constants.kcal,
523    ...       temperature=300.0, scale="cm3 particle-1 s-1")
524    array([5.14e-13])
525    >>> get_k(model.scheme, model.compounds,
526    ...       bias=np.array([0.0, 0.0, -1.4, 0.0, 0.0]) * constants.kcal,
527    ...       temperature=300.0, scale="cm3 particle-1 s-1")
528    array([1.1e-12])
529    """
530    qrrho_enthalpy, qrrho_entropy = _check_qrrho(qrrho)
531    scheme = rx.core._check_scheme(scheme)
532    if compounds is not None:
533        compounds = rx.io._check_compounds(compounds)
534    if delta_freeenergies is None:
535        assert compounds is not None, "compounds could not be inferred"
536        freeenergies = get_freeenergies(
537            compounds,
538            bias=bias,
539            qrrho=(qrrho_enthalpy, qrrho_entropy),
540            # NOTE(schneiderfelipe): ensure we get rate constants in M-1 s-1.
541            # TODO(schneiderfelipe): this strategy will have to change
542            # somewhat when we improve solvation entropy models.
543            environment="solvent",
544            temperature=temperature,
545            pressure=pressure,
546        )
547
548        # TODO(schneiderfelipe): log the contribution of reaction symmetry
549        delta_freeenergies = rx.get_delta(
550            scheme.B, freeenergies
551        ) - temperature * rx.get_reaction_entropies(
552            scheme.B, temperature=temperature, pressure=pressure
553        )
554
555    if molecularity is None:
556        molecularity = rx.thermo.get_molecularity(scheme.A)
557
558    k = rates.eyring(
559        delta_freeenergies,
560        # NOTE(schneiderfelipe): using molecularity here messes rate constant
561        # units (by a factor of M-1 s-1 to atm-1 s-1), so we leave it as is.
562        #
563        # molecularity=molecularity,
564        temperature=temperature,
565        pressure=pressure,
566        volume=volume,
567    )
568
569    # make reaction rate constants for equilibria as close as possible to one
570    i = 0
571    while i < len(scheme.is_half_equilibrium):
572        if scheme.is_half_equilibrium[i]:
573            pair = k[i : i + 2]
574            _K = pair[0] / pair[1]
575
576            denom = pair.min()
577            if denom == 0.0:
578                logger.warning(
579                    "found half-equilibrium reaction with zero rate constant: "
580                    "skipping equilibrium normalization"
581                )
582                denom = 1.0
583
584            k[i : i + 2] = pair / denom
585            assert np.isclose(_K, k[i] / k[i + 1]), (
586                f"reaction rate constants {k[i]} and {k[i + 1]} for "
587                "equilibria could not be made to match the expected "
588                f"equilibrium constant value {_K}"
589            )
590
591            # loop over pairs of equilibria
592            i += 1
593        i += 1
594
595    logger.info(
596        "(classical) reaction rate constants: "
597        f"{', '.join([f'{v:7.3g}' for v in k])} atm⁻ⁿ⁺¹·s⁻¹"
598    )
599    if tunneling not in {"none", None}:
600        if compounds is not None:
601            kappa = get_kappa(
602                scheme,
603                compounds,
604                method=tunneling,
605                qrrho=qrrho_enthalpy,
606                temperature=temperature,
607            )
608            k *= kappa
609        else:
610            # TODO(schneiderfelipe): when get_kappa accept deltas, this will
611            # be probably unnecessary.
612            logger.warning(
613                "assuming unitary tunneling coefficients due to incomplete "
614                "compound data"
615            )
616        logger.info(
617            "(tunneling) reaction rate constants: "
618            f"{', '.join([f'{v:7.3g}' for v in k])} atm⁻ⁿ⁺¹·s⁻¹"
619        )
620
621    # TODO(schneiderfelipe): ensure diffusional limit for reactions in

713        energies = get_enthalpies(compounds, qrrho=qrrho, temperature=0.0)
714        delta_forward = rx.get_delta(scheme.B, energies)  # B - A
715        delta_backward = delta_forward - rx.get_delta(
716            scheme.A, energies
717        )  # B - C == B - A - (C - A)
718

720    for i, ts in enumerate(
721        rx.get_transition_states(scheme.A, scheme.B, scheme.is_half_equilibrium)
722    ):

727            vibfreq = compounds[transition_state].vibfreqs[0]
728

730                with warnings.catch_warnings():
731                    warnings.filterwarnings("error")
732                    try:
733                        kappa = tunnel.eckart(
734                            vibfreq,
735                            delta_forward[i],
736                            delta_backward[i],
737                            temperature=temperature,
738                        )
739                    except RuntimeWarning as e:
740                        logger.warning(f"using Wigner tunneling correction: {e}")
741                        kappa = tunnel.wigner(vibfreq, temperature=temperature)
742            elif method == "wigner":
743                kappa = tunnel.wigner(vibfreq, temperature=temperature)
744            elif method in {"none", None}:
745                kappa = 1.0
746            else:
747                raise ValueError(f"unavailable method: '{method}'")
748
749            kappas.append(kappa)

752    # somewhere else?
753    vec_kappas = np.asarray(kappas).flatten()
754    logger.info(
755        "(quantum) tunneling coefficients: "
756        f"{', '.join([f'{kappa:7.3g}' for kappa in vec_kappas])}"
757    )
